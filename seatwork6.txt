#include <iostream>
using namespace std;

int main (){
    double num1, num2;
    char op;
    
    cout << "Enter Operation (+,-,*,/): ";
    cin >> op;
    cout << "Enter first number: ";
    cin >> num1;
    cout << "Enter second number: ";
    cin >> num2;
    
    switch(op){
        case '+':
          cout << "Result: " << (num1+num2);
        break; 
        case '*':
          cout << "Result: " << (num1*num2);
        break;
        case '-':
          cout << "Result: " << (num1-num2);
        break;
        case '/':
          cout << "Result: " << (num1/num2);
        break;
        
        default:
          cout << "Invalid operation!";
        break;
        
    }
}
